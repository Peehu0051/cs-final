    <script>
        let cart = JSON.parse(localStorage.getItem('cart')) || [];

        function addToCart(productName, price, size, shipping) {
            const item = { name: productName, price: parseFloat(price.slice(1)), size, shipping };
            cart.push(item);
            alert("Item added to cart!");
        }

        function showHome() {
            document.getElementById('content').innerHTML = `
                <div style="text-align: center; padding: 20px;">
                    <h2>Welcome to TrendFusion</h2>
                    <p>Explore the beauty of sustainable and stylish clothing. Our unique collections blend fashion with responsibility.</p>
                </div>`;
        }

        function showShop() {
            document.getElementById('content').innerHTML = document.querySelector('.product-gallery').outerHTML;
        }

        function showContact() {
            document.getElementById('content').innerHTML = `
                <div style="text-align: center; padding: 20px;">
                    <h2>Contact Us</h2>
                    <p>Phone: 123-456-7890</p>
                    <p>Instagram: @TrendFusion</p>
                </div>`;
        }

        function showProductPage(name, imageUrl, price, sizes, description) {
            document.getElementById('content').innerHTML = `
                <div class="product-detail">
                    <img src="${imageUrl}" alt="${name}">
                    <h2>${name}</h2>
                    <p><strong>Price:</strong> ${price}</p>
                    <p>${description}</p>
                    <div class="size-selector">
                        <label for="sizes">Select Size:</label>
                        <select id="sizes">
                            ${sizes.split(', ').map(size => `<option value="${size}">${size}</option>`).join('')}
                        </select>
                    </div>
                    <div class="shipping-selector">
                        <label for="shipping">Select Shipping:</label>
                        <select id="shipping">
                            <option value="Normal">Normal (3-5 days)</option>
                            <option value="Express">Express (1-2 days)</option>
                        </select>
                    </div>
                    <button onclick="addToCart('${name}', '${price}', document.getElementById('sizes').value, document.getElementById('shipping').value)">Add to Cart</button>
                </div>`;
        }

        function showCart() {
            if (cart.length === 0) {
                document.getElementById('content').innerHTML = '<p style="text-align: center;">Your cart is empty.</p>';
                return;
            }

            let cartHTML = '<div class="cart">';
            let total = 0;

            cart.forEach((item, index) => {
                total += item.price;
                cartHTML += `
                    <div class="cart-item">
                        <span onclick="showProductPage('${item.name}', findImageForProduct('${item.name}'), '$${item.price.toFixed(2)}', 'S, M, L, XL', 'Product details here')">${item.name} (${item.size}, ${item.shipping})</span>
                        <span>$${item.price.toFixed(2)}</span>
                        <button onclick="removeFromCart(${index})">Remove</button>
                    </div>`;
            });

            cartHTML += `
                <div class="cart-summary">
                    <p><strong>Total:</strong> $${total.toFixed(2)}</p>
                    <button onclick="placeOrder()">Place Order</button>
                </div>
            </div>`;

            document.getElementById('content').innerHTML = cartHTML;
        }

        function removeFromCart(index) {
            cart.splice(index, 1);
            showCart();
        }

        function placeOrder() {
            alert('Thank you for your order!');
            cart = [];
            showCart();
        }

        function findImageForProduct(name) {
            const products = {
                'Eco-Friendly T-Shirt': 'https://www.cottonheritage.com/catImg/WAMHIRES/lc1025100_4.jpg',
                'Recycled Denim Jeans': 'https://lsco.scene7.com/is/image/lsco/A91500002-dynamic1-pdp',
                'Bamboo Sunglasses': 'https://encrypted-tbn0.gstatic.com/images?q=tbn:ANd9GcQOLI7A6zXOgYrdc3AlZgfHRCalP9qX5lhkcg&s',
                'Recycled Tote Bag': 'https://eswbeauty.com/cdn/shop/products/IMG_9665_square.jpg?v=1663146642&width=480',
                'Organic Hoodie': 'https://rawganique.com/cdn/shop/files/surf-city-hoodie-20231027-DSCF3335-3_1024x1024@2x.jpg?v=1698690250'
            };
            return products[name] || '';
        }
